##### Router 1 ######
enable
configure terminal
interface serial 0/0/0
ip address 192.168.10.10 255.255.255.0
interface serial 0/0/1
ip address 10.0.0.1 255.0.0.0
interface serial 0/0/3
ip address 20.0.0.1 255.0.0.0
router rip
version 2
no auto-summary
network 192.168.1.0
network 10.0.0.0
network 20.0.0.0
exit
